def short_udr(matrix):
    shortest = [matrix[row][0] for row in range(len(matrix))]

    for j in range(1, len(matrix)):
        column = [matrix[row][j] for row in range(len(matrix))]

        shortest = [ column[row] + min([shortest[last_row] + 
            sum(column[last_row:row:(1 if last_row <= row else -1)])
                for last_row in range(len(matrix))])
            for row in range(len(matrix))]

    return min(shortest)
    
def matr(txt):
    f = open ( txt , 'r')
    l = [ map(int,line.split(',')) for line in f ]
    f.close
    return l
